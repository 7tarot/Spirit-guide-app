<script>
  const a = document.getElementById('a');
  document.getElementById('sel').addEventListener('change', e=>{
    a.pause(); a.removeAttribute('src');
    if(!e.target.value) return;
    a.src = e.target.value; a.volume = 0.35;
    a.play().catch(()=>{});
  });
  document.getElementById('stop').onclick=()=>{ a.pause(); a.removeAttribute('src'); };
</script>
